# PUT
curl -i -H "Content-Type: application/json" -X PUT -d '{"name":"solomon", "email":"ssbothwell@gmail.com", "phone":"3238047139"}' http://localhost:5000

curl -i -H "Content-Type: application/json" -H "Authorization: Bearer $ACCESS" -X PUT -d '{"name":"solomon", "email":"ssbothwell@gmail.com", "phone":"3238047139"}' http://localhost:5000

# GET
curl -i -H "Content-Type: application/json" -X GET -d '{"memberID": "1"}' http://localhost:5000

curl -i -H "Content-Type: application/json" -H "Authorization: Bearer $ACCESS" -X get -d '{"memberID": "1"}' http://localhost:5000
